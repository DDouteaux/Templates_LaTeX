%
% Title format of numbered subsubsections. Based on applySectionFormat method.
%
\titleformat{\subsubsection}
  {\fontsize{13pt}{0pt}\selectfont}
  {}
  {0em}
  {%
    \applySectionFormat{\subsubsectionStyle}%
    {\sectionNumberFormating{\subsubsectionNumberStyle}{\subsubsectionNumberSize}{\subsubsectionNumberColor}{\thesubsubsection}}%
    {\sectionSeparatorFormating{\subsubsectionSeparatorStyle}{\subsubsectionSeparatorSize}{\subsubsectionSeparatorColor}{\subsubsectionSeparator}}%
    {\sectionNameFormating{\subsubsectionNameStyle}{\subsubsectionNameSize}{\subsubsectionNameColor}{#1}}%
  }


%
% Title format of numbered subsubsections. Based on applySectionFormat method.
%
\titleformat{name=\subsubsection,numberless}
  {\fontsize{13pt}{0pt}\selectfont}
  {}
  {0em}
  {%
    \applySectionFormat{\subsubsectionStyle}%
    {}%
    {}%
    {\sectionNameFormating{\subsubsectionNameStyle}{\subsubsectionNameSize}{\subsubsectionNameColor}{#1}}%
  }


%
% Subsubsection numbering style
%
\renewcommand\thesubsubsection{\arabic{section}.\arabic{subsection}.\arabic{subsubsection}}

%%
%% Variables exposed by this file 
%%
\newcount\loopDocumentPartCounter
\newbool{mainContentDisplayed}


%%
%% Loop commands to print document part
%%
\newcommand{\loopDocumentPartNumber}{%
  \loopDocumentPartCounter-1                           % initialize the loop counter
  \loop\ifnum\loopDocumentPartCounter < 1000           % test if the loop counter is < 1000
  [\displayDocumentPartNumber{\loopDocumentPartCounter}~~\number\loopDocumentPartCounter] % do our actions
  %[\displayCustomDocumentPartNumber{\loopDocumentPartCounter}~~\number\loopDocumentPartCounter] % do our actions
  \advance\loopDocumentPartCounter by 1                % increase loop counter by one 
  \repeat                                              % repeat the loop
}

\newcommand*{\displayDocumentPartNumber}[1]{%
  % First check if we display main content
  \ifnum\whereMainContent=#1
    \displayMainContent
    \booltrue{mainContentDisplayed}
  \fi
  \ifbool{mainContentDisplayed}{%
    % Main content already displayed
    \ifnum\whereConclusion=#1\displayConclusion\fi
    \ifnum\whereSummaries=#1\displaySummaries\fi
    \ifnum\whereAppendix=#1\displayAppendix\fi
    \ifnum\whereLastPage=#1\displayLastPage\fi
  }{%
    % Main content not displayed
    \ifnum\whereTitlePage=#1\displayTitlePage\fi
    \ifnum\whereAbstracts=#1\displayAbstracts\fi
    \ifnum\whereIntroduction=#1\displayIntroduction\fi
  }
  % These parts can be displayed before or after main content.
  \ifnum\whereTOC=#1\displayTOC\fi
  \ifnum\whereLOTLOF=#1\displayLOTLOF\fi
  \ifnum\whereThanks=#1\displayThanks\fi
  \ifnum\whereReferences=#1\displayReferences\fi
  \ifnum\whereIndex=#1\displayIndex\fi
}

\newcommand*{\displayCustomDocumentPartNumber}[1]{%

}

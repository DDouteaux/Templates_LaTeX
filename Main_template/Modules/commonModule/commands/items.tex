
%% ================================================================
%%
%% Custom bicolored itemize environment
%%
%% ================================================================

% Params :
%    1. Symbol to use. For possible keys see styles/symbols/symbols__item.
%    2. Main symbol color (if able). For possible keys see styles/colors/*.
%    3. Second symbol color (if able). For possible keys see styles/colors/*.
%    4. Base size for bullets. Value to store in lengthBulletHeight. If empty, no update is carried.
%    5. Baseline shift of figure in percentage of lengthBulletHeight (100% <=> 1). 
%
% If you don't want to update a parameter, leave its value blank!
\makeatletter
\newenvironment{biColorItemize}[5]{%
	\ifstrempty{#4}{}{\updateLengthBulletHeight{#4}}
	\ifstrempty{#2}{}{\renewcommand{\itemBulletsColor}{#2}}
  	\ifstrempty{#3}{}{\renewcommand{\itemBulletsSideColor}{#3}}
  	\ifstrempty{#5}{}{\updateLengthBulletBaselineShift{#5}}
  	\ifstrempty{#1}{}{\renewcommand\labelitemi{#1}}  	
	\begin{itemize}
}{%
	\end{itemize}
}
\makeatother

%
% 2 - Placing the different parts of the document
%
% The outcomes are self-explanatory and will be detailed line by line.
% Some important remarks :
%        - if you chose to have a title page it will always be your first page
%        - if you chose to have a last page page it will always be the last one
%        - LOF and LOT will always be bind, but the order is up to you
%        - abstract and abstract native will always be bind, but the order is up to you
%        - summary and summary native will always be bind, but the order is up to you
%        - 0 will always be the first page after the title page
%        - 999 will always be the last page before last page
%        - abstract and introduction will always be before main matter
%        - conclusion and summary womm always be after the main matter
%
\newcommand\whereTitlePage{-1}     % Always the first page
\newcommand\whereTOC{0}            % Scope 1-999
\newcommand\whereLOTLOF{1}         % Scope 1-999
\newcommand\whereAbstracts{2}      % Scope 1-999
\newcommand\whereThanks{3}         % Scope 1-999
\newcommand\whereIntroduction{4}   % Scope 1-999
\newcommand\whereMainContent{5}    % Scope 1-999
\newcommand\whereConclusion{6}     % Scope 1-999
\newcommand\whereSummaries{7}      % Scope 1-999
\newcommand\whereReferences{8}     % Scope 1-999
\newcommand\whereIndex{9}          % Scope 1-999
\newcommand\whereAppendix{10}      % Scope 1-999
\newcommand\whereLastPage{999}     % Always the last page

%% ================================================================
%%
%% PART 1 : Some explanations about this file
%%
%% ================================================================

% Each parameter used will be passed when the icon is called. The parameters mainly
% concerns the size and colors of the icons.
%
% Default printing commands are also given in order to user these symbols without
% any configuration (or only color). As the main purpose of these symbols is to be
% used inline, these default commands will use the \lengthTextHeight size.



%% ================================================================
%%
%% PART 2 : Symbol definition with description
%%
%% ================================================================

%%
%% Circle symbol
%%
%% Params :
%%   - Fill color of the symbol
%%   - Draw color of the symbol
%%   - Size of reference for the symbol 
%%   - Baseline offset for the symbol
\newcommand{\symbolCircleGeneral}[4]{\tikz[baseline={([yshift=#4]current bounding box.center)}]{ \draw [color=#2, fill=#1] (0,0) circle (.5#3); }}

%%
%% Square symbol
%%
%% Params :
%%   - Fill color of the symbol
%%   - Draw color of the symbol
%%   - Size of reference for the symbol 
%%   - Baseline offset for the symbol
\newcommand{\symbolSquareGeneral}[4]{\tikz[baseline={([yshift=#4]current bounding box.center)}]{ \draw [color=#2, fill=#1] (0,0) rectangle (#3,#3); }}

%%
%% Diamond symbol
%%
%% Params :
%%   - Fill color of the symbol
%%   - Draw color of the symbol
%%   - Size of reference for the symbol 
%%   - Baseline offset for the symbol
\newcommand{\symbolDiamondGeneral}[4]{\tikz[baseline={([yshift=#4]current bounding box.center)}]{ \draw [color=#2, fill=#1] (0,.5#3) -- (.5#3,#3) -- (#3,.5#3) -- (.5#3,0) --cycle; }}

%%
%% Target symbol
%%
%% Params :
%%   - Color of the exterior ring
%%   - Color of the inner ring
%%   - Size of reference for the symbol 
%%   - Baseline offset for the symbol
\newcommand{\symbolTargetGeneral}[4]{\tikz[baseline={([yshift=#4]current bounding box.center)}]{ \fill [#1] (0,0) circle (.5#3); \fill [white] (0,0) circle (.35#3); \fill [#2] (0,0) circle (.15#3); }}



%% ================================================================
%%
%% PART 3 : Default printing of symbols
%%
%% ================================================================

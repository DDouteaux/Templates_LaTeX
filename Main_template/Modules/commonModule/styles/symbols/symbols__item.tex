%%---------------------------------------------------------------%%
%                                                                 %
%                     Symbol for item bullets                     %
%                                                                 %
%      List of symbols that will be displayed as item bullets.    %
%                                                                 %
%%---------------------------------------------------------------%%

%% ================================================================
%%
%% Variable used
%%
%% ================================================================

% Theses symbols use the following variables :
%    - lengthBulletHeight : expected height for the bullets
%    - lengthBulletBaselineShift : baseline shift for the bullets
%    - itemBulletsColor : main color for the bullets
%    - itemBulletsSideColor : side color for bicolored bullets



%% ================================================================
%%
%% Bullets with predefined chars
%%
%% ================================================================

\newcommand{\itemFFemtoBullet}{$\vcenter{\hbox{\tiny\textcolor{\itemBulletsColor}{$\cdot$}}}$}
\newcommand{\itemTinyBullet}{$\vcenter{\hbox{\tiny\textcolor{\itemBulletsColor}{$\bullet$}}}$}



%% ================================================================
%%
%% Tikz drawn symbols monocolored
%%
%% ================================================================

\newcommand{\itemCircleGeneral}{\symbolCircleGeneral{white}{\itemBulletsColor}{\lengthBulletHeight}{\lengthBulletBaselineShift}}
\newcommand{\itemDiskGeneral}{\symbolCircleGeneral{\itemBulletsColor}{\itemBulletsColor}{\lengthBulletHeight}{\lengthBulletBaselineShift}}
\newcommand{\itemSquareGeneral}{\symbolSquareGeneral{white}{\itemBulletsColor}{\lengthBulletHeight}{\lengthBulletBaselineShift}}
\newcommand{\itemSquareFillGeneral}{\symbolSquareGeneral{\itemBulletsColor}{\itemBulletsColor}{\lengthBulletHeight}{\lengthBulletBaselineShift}}
\newcommand{\itemDiamondGeneral}{\symbolDiamondGeneral{white}{\itemBulletsColor}{\lengthBulletHeight}{\lengthBulletBaselineShift}}
\newcommand{\itemDiamondFillGeneral}{\symbolDiamondGeneral{\itemBulletsColor}{\itemBulletsColor}{\lengthBulletHeight}{\lengthBulletBaselineShift}}
\newcommand{\itemTargetGeneral}{\symbolTargetGeneral{\itemBulletsColor}{\itemBulletsColor}{\lengthBulletHeight}{\lengthBulletBaselineShift}}



%% ================================================================
%%
%% Tikz drawn symbols bicolored
%%
%% ================================================================

\newcommand{\itemCircleBiGeneral}{\symbolCircleGeneral{\itemBulletsSideColor}{\itemBulletsColor}{\lengthBulletHeight}{\lengthBulletBaselineShift}}
\newcommand{\itemSquareBiGeneral}{\symbolSquareGeneral{\itemBulletsSideColor}{\itemBulletsColor}{\lengthBulletHeight}{\lengthBulletBaselineShift}}
\newcommand{\itemDiamondBiGeneral}{\symbolDiamondGeneral{\itemBulletsSideColor}{\itemBulletsColor}{\lengthBulletHeight}{\lengthBulletBaselineShift}}
\newcommand{\itemTargetBiGeneral}{\symbolTargetGeneral{\itemBulletsSideColor}{\itemBulletsColor}{\lengthBulletHeight}{\lengthBulletBaselineShift}}
